cmake_minimum_required(VERSION 3.10)
project(GraphicsEngine VERSION 1.0.0)

find_package(OpenGL REQUIRED)

add_executable(GraphicsEngine 
    src/main.h
    src/main.cpp
    src/glad.c
)

target_include_directories(GraphicsEngine PRIVATE dependencies)

target_link_directories(GraphicsEngine PRIVATE dependencies/GLFW3)
target_link_libraries(GraphicsEngine PRIVATE glfw3 OpenGL::GL)
